## Свойство объекта `@property`

```python
class MyClass:
	def __init__(self, n):
		self.__value = n
	@property
	def get_value(self):
		return self.__value
	def info(self):
		print(f"value = {self.get_value}")

obj = MyClass(10)
print(obj.__value) # ошибка!
print(obj.get_value()) # ошибка!
print(obj.get_value) # 10
```

```python
class MyClass:
	def __init__(self, x, y):
		self.__x = x
		self.__y = y
	def get_x(self):
		return self.__x
	def set_x(self, x):
		self.__x = x
	x = property(get_x, set_x)

obj = MyClass(10, 20)
obj.x = 30 # вызывается set_x()
a = obj.x # вызывается get_x()
```
